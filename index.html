A simple replacement for Javascript's alerts

Responsive, zero dependencies (vanilla), lightweight (2.8kB)

Usage

  • load
    <script src="BabNotifier.min.js"></script>
    <link rel="stylesheet" href="BabNotifier.min.css">
  • initialize
    const notifier = new BabNotifier();

Examples

  • notifier.notify({
      title: 'Hey there',
      body: 'This will stay open untill clicked'
    });
  • notifier.notify({
      title: 'Hey there',
      icon: 'success',
      body: 'This will stay open untill clicked'
    });
  • notifier.notify({
      title: 'Hey there!',
      icon: 'error',
      body: 'This will stay open untill clicked'
    });
  • notifier.notify({
      title: 'Hey there',
      html:
        'This popup will close after 3 seconds. ' +
        'at least that is what it should do acording to the code, ' +
        'and I wrote the code, so I have an idea what it should do. ' +
        'Oh, <a href="https://www.google.com">this is a link</a>.',
      timer: 6000
    });
  • notifier.notify({
      type: 'toaster',
      title: 'Settings were saved'
    });
  • notifier.notify({
      type: 'toaster',
      icon: 'success',
      title: 'Settings were saved'
    });
  • notifier.notify({
      type: 'toaster',
      icon: 'error',
      title: 'Something went wrong'
    });
